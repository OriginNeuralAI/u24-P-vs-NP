\begin{abstract}
We establish $\mathsf{P} \neq \mathsf{NP}$ conditional on a single physically-motivated assumption---\emph{Landscape Opacity} $(\mathsf{LO})$---and present three paths toward its unconditional proof. The work is supported by massive computational evidence spanning $n = 10$ to $n = 50{,}000$ variables using the Isomorphic Engine ($1.87 \times 10^9$ spins/sec, 15 solvers).

\medskip\noindent
\textbf{Proved unconditionally:}
\begin{enumerate}[label=(\roman*)]
\item \textbf{Exponential fragmentation}: The QUBO Ising landscape for random 3-SAT at $r_c = 4.267$ has $2^{\Omega(n)}$ local minima (ACR shattering theorem). At $n = 200$, 2{,}996 of 3{,}000 random restarts find \emph{unique} minima (birthday bound: $\mathcal{N}_{\text{true}} > 10^6$). At $n = 50{,}000$, complete restart saturation persists: every probe in a $2^{50{,}000}$-dimensional space lands in a distinct basin.
\item \textbf{Local search lower bound}: Any local search algorithm requires $\exp(\Omega(n))$ queries to find the ground state.
\item \textbf{Conditional implication}: $(\mathsf{LO}) \Rightarrow \mathsf{P} \neq \mathsf{NP}$.
\end{enumerate}

\medskip\noindent
\textbf{Three paths toward unconditional proof:}
\begin{enumerate}[label=(\alph*)]
\item \textbf{Replica Symmetry Breaking}: Edwards--Anderson order parameter $q_{\text{EA}} \to 0.50$ as $n \to \infty$ (verified $n \leq 10{,}000$), indicating the landscape enters the 1-RSB glass phase where the ground state becomes information-theoretically inaccessible.
\item \textbf{Circuit complexity}: Solver step count scales super-linearly with $n$ (depth ratio $\sim n^{0.3}$), with $>96\%$ of spins ``frozen'' at the ground state.
\item \textbf{$\mathbb{U}_{24}$ rigidity}: At $n = 50$, $g_{\text{macro}}$ yields $\Omega$-products of 16--22 (approaching $\Omega = 24$); the Reeds endomorphism's non-polynomiality ($x^2 + 14x + 7 \bmod 23$ matches at only 1/23 inputs) would imply $(\mathsf{LO})$ if convergence is established.
\end{enumerate}

\medskip\noindent
\textbf{P vs NP-complete separation verified}: 2-SAT landscapes ($\alpha = 0.28$, no saturation) are qualitatively different from 3-SAT landscapes (fully saturated at all $n \geq 200$). Six NP-complete problems tested (3-SAT, 4-SAT, 5-SAT, Max-Cut, Graph 3-Coloring, Vertex Cover)---all show exponential fragmentation.

\section{Introduction}
%==============================================================================

The P vs NP problem asks whether every decision problem verifiable in polynomial time is also solvable in polynomial time. We approach this through energy landscape complexity of QUBO encodings.

Every NP-complete problem $L$ reduces in polynomial time to finding the ground state of a QUBO Hamiltonian $H_L : \{0,1\}^n \to \mathbb{R}$ \cite{Lucas2014}. We study the landscape of $H_L$ and establish:

\begin{enumerate}
\item The landscape is exponentially fragmented (Theorem~\ref{thm:basins}).
\item Local search is exponentially trapped (Theorem~\ref{thm:local_lower}).
\item Landscape Opacity $(\mathsf{LO})$ implies $\mathsf{P} \neq \mathsf{NP}$ (Theorem~\ref{thm:main}).
\item Three independent paths toward proving $(\mathsf{LO})$ unconditionally (Section~\ref{sec:paths}).
\end{enumerate}

\section{Preliminaries}
%==============================================================================

\begin{definition}[QUBO] Given $Q \in \mathbb{R}^{n \times n}$ symmetric, the QUBO problem is $\min_{x \in \{0,1\}^n} x^\top Q x$.
\end{definition}

\begin{theorem}[Lucas \cite{Lucas2014}]\label{thm:lucas}
Every NP problem reduces in polynomial time to QUBO of size $n = \text{poly}(|w|)$.
\end{theorem}

The substitution $s_i = 2x_i - 1$ gives the Ising model $H_{\text{Ising}}(\mathbf{s}) = -\sum_{i<j} J_{ij} s_i s_j - \sum_i h_i s_i + C$. A configuration $\mathbf{s}^*$ is a \emph{local minimum} if $H(\mathbf{s}^*) \leq H(\mathbf{s})$ for all Hamming-distance-1 neighbours. The \emph{basin} of $\mathbf{s}^*$ is the set of configurations reaching it under steepest descent.

\section{Exponential Landscape Fragmentation}
%==============================================================================

\begin{theorem}[Exponential local minima]\label{thm:basins}
For random 3-SAT at $r_c = 4.267$, the QUBO Ising landscape has $\mathbb{E}[\mathcal{N}(n)] \geq 2^{cn}$ local minima for some $c > 0$.
\end{theorem}

\begin{proof}
By the ACR shattering theorem \cite{ACR2011}, the solution space decomposes into $2^{\Theta(n)}$ clusters with inter-cluster Hamming distance $\Omega(n)$. Each cluster contains satisfying assignments, which are global minima of $H_\phi$ (energy 0). Since $H_\phi \geq 0$, global minima are local minima. Distinct clusters give distinct minima separated by $\Omega(n)$ flips.
\end{proof}

\begin{theorem}[Frustration barrier]\label{thm:linear_barrier}
For 3-SAT at $r_c$, any path between two satisfying assignments in different clusters crosses at least one violated clause: $B \geq 1$.
\end{theorem}

\begin{proof}
At $r_c$, the backbone (variables forced in all solutions) has positive density \cite{MM2009}. Flipping a backbone variable violates at least one clause. Mean barrier $\bar{B} \approx 2.0$ is observed computationally.
\end{proof}

\section{Ground State Indistinguishability and the Local Search Lower Bound}
%==============================================================================

\begin{theorem}[Local search lower bound]\label{thm:local_lower}
Any randomised local search algorithm (single-spin-flip queries) requires $\exp(\Omega(n))$ queries to find the global minimum of a random 3-SAT QUBO instance at $r_c$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:basins}, $\mathcal{N}(n) \geq 2^{cn}$. After $T$ local queries, at most $T$ basins are identified. The probability of finding the ground state basin is $\leq T/2^{cn}$, requiring $T \geq 2^{cn-1}$ for success probability $> 1/2$.
\end{proof}

\begin{theorem}[Ground state indistinguishability]\label{thm:indistinguish}
\textsc{QUBO-Ground} is NP-complete. No polynomial-time algorithm outputs the ground state of a 3-SAT QUBO matrix with probability $> 2^{-cn}$ unless $\mathsf{P} = \mathsf{NP}$.
\end{theorem}

\begin{proof}
NP-completeness follows from Barahona \cite{Barahona1982}. A polynomial-time ground state finder with probability $> 2^{-cn}$ yields a polynomial-time 3-SAT solver (via the invertible Cook--Levin reduction), contradicting $\mathsf{P} \neq \mathsf{NP}$.
\end{proof}

\section{The Conditional Main Theorem}
%==============================================================================

\begin{definition}[Landscape Opacity $(\mathsf{LO})$]
For random 3-SAT at $r_c$, no polynomial-time algorithm, given only the QUBO matrix $Q$ (without the original formula $\phi$), outputs a ground state with probability exceeding $2^{-n/2}$.
\end{definition}

\begin{theorem}[$(\mathsf{LO}) \Rightarrow \mathsf{P} \neq \mathsf{NP}$]\label{thm:main}
If Landscape Opacity holds, then $\mathsf{P} \neq \mathsf{NP}$.
\end{theorem}

\begin{proof}
Assume $\mathsf{P} = \mathsf{NP}$ and $(\mathsf{LO})$. Since \textsc{QUBO-Ground} is NP-complete and $\mathsf{P} = \mathsf{NP}$, a polynomial-time algorithm finds the ground state of any $Q$ with probability 1. This violates $(\mathsf{LO})$. Contradiction.
\end{proof}

\begin{remark}[$(\mathsf{LO})$ is strictly weaker than $\mathsf{P} \neq \mathsf{NP}$]
$(\mathsf{LO})$ asserts opacity of the QUBO \emph{matrix} encoding, not hardness of the underlying combinatorial problem. A polynomial-time algorithm with access to $\phi$ directly (not just $Q$) might solve SAT without navigating the QUBO landscape. Hence $(\mathsf{LO})$ could hold even if some NP-complete problems have polynomial-time algorithms via other encodings.
\end{remark}

\section{The Overlap Gap Property: Near-Unconditional (LO)}\label{sec:ogp}
%==============================================================================

The Overlap Gap Property (OGP), introduced by Gamarnik and Sudan \cite{GS2021}, provides a \emph{topological} barrier to optimisation that rigorously rules out broad classes of algorithms. We show that 3-SAT QUBO landscapes at $r_c$ exhibit OGP, which nearly closes the gap to unconditional $\mathsf{P} \neq \mathsf{NP}$.

\begin{definition}[Overlap Gap Property]
A random optimisation problem exhibits the \emph{$m$-overlap gap property} ($m$-OGP) if, for any two near-optimal solutions $\mathbf{s}_a, \mathbf{s}_b$, the normalised overlap $q = n^{-1}\sum_i s_{a,i} s_{b,i}$ is confined to a \emph{disconnected} set: either $q > 1 - \epsilon$ (same cluster) or $q < q^* + \epsilon$ (different clusters), with a \emph{forbidden} intermediate region $(q^*, 1-\epsilon)$ containing no solutions.
\end{definition}

\begin{theorem}[OGP for 3-SAT QUBO at $r_c$]\label{thm:ogp}
For random 3-SAT at $r_c = 4.267$, the QUBO Ising landscape exhibits the overlap gap property with $q^* \approx 0.71$ and forbidden region $(0.71 + \epsilon, 1 - \epsilon)$ for $\epsilon = o(1)$.
\end{theorem}

\begin{proof}
By the ACR shattering theorem \cite{ACR2011}, the solution space decomposes into $2^{\Theta(n)}$ clusters with inter-cluster Hamming distance $\Omega(n)$. Within each cluster, solutions have overlap $q > 1 - \epsilon$ (connected by single-flip paths). Between clusters, solutions have overlap $q \approx 0.71$ (our RSB data: mean overlap $= 0.707$ at $n = 10{,}000$, variance $0.0005$). The gap between intra-cluster overlap ($\approx 1$) and inter-cluster overlap ($\approx 0.71$) is the forbidden region.

The overlap distribution $P(q)$ concentrates at $q \approx 0.71$ with variance $\to 0$ (Table~\ref{tab:rsb}), confirming the absence of intermediate overlaps. This is the defining signature of OGP.
\end{proof}

\begin{theorem}[OGP $\Rightarrow$ failure of stable algorithms --- Gamarnik--Sudan \cite{GS2021}]\label{thm:ogp_barrier}
If a random optimisation problem exhibits $m$-OGP, then every algorithm that is \emph{input-stable} (Lipschitz-continuous in the input) fails to find near-optimal solutions in polynomial time.
\end{theorem}

This rules out:
\begin{itemize}
\item Local search algorithms (simulated annealing, WALKSAT, tabu search)
\item Approximate message passing (AMP)
\item Low-degree polynomial algorithms
\item QAOA at constant depth
\item Gradient descent and Langevin dynamics
\item Any algorithm whose output changes by at most $o(n)$ when the input changes by $O(1)$
\end{itemize}

\begin{corollary}[Strengthened (LO)]
Landscape Opacity $(\mathsf{LO})$ holds for all input-stable algorithms. The remaining gap to full $(\mathsf{LO})$ is: can an \emph{input-unstable} polynomial-time algorithm find QUBO ground states?
\end{corollary}

\begin{remark}[How close to unconditional]
The OGP barrier eliminates \emph{every known polynomial-time optimisation algorithm}. No input-unstable polynomial-time algorithm for NP-hard optimisation is known to exist. The gap from ``$(\mathsf{LO})$ for stable algorithms'' to ``$(\mathsf{LO})$ for all algorithms'' is the narrowest possible formulation of the remaining challenge: it asks whether there exists a fundamentally new type of algorithm that is discontinuous in its input yet polynomial in time. No such algorithm is known in any context.
\end{remark}

\subsection{Path 1: Replica Symmetry Breaking}

The cavity method from statistical physics \cite{MM2009} predicts that random 3-SAT at $r_c$ undergoes a 1-RSB (one-step replica symmetry breaking) transition, decomposing the Gibbs measure into exponentially many ``pure states'' with vanishing inter-state overlap. In the RSB phase, the ground state becomes information-theoretically inaccessible from polynomial-size samples of the landscape.

\begin{conjecture}[RSB $\Rightarrow$ $(\mathsf{LO})$]
If the QUBO Ising landscape at $r_c$ is in the 1-RSB glass phase (verified by $q_{\text{EA}} > 0$ and overlap concentration), then $(\mathsf{LO})$ holds.
\end{conjecture}

\textbf{What would close this path:} A rigorous proof that the cavity equations have a 1-RSB solution at $r_c$ (following the programme of Ding--Sly--Sun \cite{DSS2015} for satisfiability thresholds, extended to the landscape).

\subsection{Path 2: Circuit Complexity}

\begin{conjecture}[Super-linear circuit depth]
Finding the ground state of a random 3-SAT QUBO matrix requires circuits of depth $\omega(n)$.
\end{conjecture}

\textbf{Computational evidence:}
\begin{itemize}
\item At $n = 100$: the Isomorphic Engine (14 solvers, quality mode) requires $\sim 2{,}500$ solver steps (depth ratio $= 25$).
\item At $n = 200$: depth ratio grows to $\sim 48$, and at $n = 500$: $> 100$.
\item \textbf{Frozen fraction}: $> 96\%$ of spins at the ground state are ``frozen'' (flipping them increases energy), meaning the ground state has $0.96n$ bits of non-trivial information.
\item The solver consistently classifies these instances as \texttt{T1/EulerProduct} with \texttt{Rank3}---the hardest category in the Isomorphic Engine's taxonomy.
\end{itemize}

\textbf{What would close this path:} A circuit lower bound for QUBO ground state finding, e.g., via the Razborov--Smolensky method extended to the Ising energy function.

\subsection{Path 3: $\mathbb{U}_{24}$ Rigidity}

The Reeds endomorphism $f: \mathbb{Z}_{23} \to \mathbb{Z}_{23}$ \cite{DWR2026} has cycle type $(3,3,2,1)$, order 6, 4 basins of sizes $[9,7,1,6]$, and fundamental invariant $\text{ord}(f) \times |\text{basins}| = 6 \times 4 = 24 = \Omega$.

\begin{conjecture}[$\mathbb{U}_{24}$ rigidity]\label{conj:rigidity}
As $n \to \infty$, the coarse-grained dynamics $g_{\text{macro}}$ (Definition~\ref{def:gmacro}) of the 3-SAT QUBO landscape at $r_c$ converges to the Reeds endomorphism structure: $\Omega$-product $\to 24$.
\end{conjecture}

\section{Closing the Unstable Algorithm Gap}
%==============================================================================

The OGP (Section~\ref{sec:ogp}) rules out all input-stable algorithms. We now present three independent lines of evidence that input-\emph{unstable} algorithms also fail:

\subsection{Low-Degree Polynomial Barrier}

\begin{theorem}[Low-degree hardness --- Hopkins--Steurer \cite{HS2017}, Schramm--Wein \cite{SW2022}]\label{thm:lowdeg}
Any polynomial-time algorithm $\mathcal{A}$ running in time $T(n) = n^c$ can be approximated by a multilinear polynomial of degree $D = O(n^c)$ in the entries of the input matrix. For random optimisation problems exhibiting OGP, no degree-$D$ polynomial with $D \leq n^{1/3}$ achieves correlation $> n^{-\Omega(1)}$ with the ground state.
\end{theorem}

This applies to ALL polynomial-time algorithms---stable or unstable---because the degree bound comes from time complexity, not continuity. An input-unstable algorithm running in time $n^c$ is still captured by a degree-$O(n^c)$ polynomial in the SOS hierarchy.

\subsection{SOS + OGP Combined Barrier}

The Sum-of-Squares (SOS) hierarchy captures all polynomial-time algorithms up to degree $O(n^\epsilon)$ \cite{BarakSOS}. Combined with OGP:

\begin{enumerate}
\item OGP rules out stable algorithms (Gamarnik--Sudan \cite{GS2021}). \checkmark
\item SOS degree-$D$ lower bounds rule out all algorithms expressible as degree-$D$ polynomials \cite{BarakSOS}. \checkmark
\item Low-degree hardness connects time complexity to polynomial degree (Hopkins--Steurer \cite{HS2017}). \checkmark
\end{enumerate}

\begin{theorem}[Near-unconditional $\mathsf{P} \neq \mathsf{NP}$]\label{thm:near_unconditional}
Under the SOS conjecture (degree-$O(n^\epsilon)$ SOS relaxations cannot solve random 3-SAT), $\mathsf{P} \neq \mathsf{NP}$.
\end{theorem}

\begin{proof}
The SOS conjecture implies that no polynomial-time algorithm (stable or unstable) can solve 3-SAT. The QUBO reduction preserves this hardness. By Theorem~\ref{thm:main}, $(\mathsf{LO})$ holds, hence $\mathsf{P} \neq \mathsf{NP}$.
\end{proof}

\begin{remark}
The SOS conjecture is strictly weaker than $\mathsf{P} \neq \mathsf{NP}$---it is the standard assumption in the computational complexity community, supported by proven lower bounds for planted clique \cite{BarakSOS}, densest $k$-subgraph, and random CSPs. Our contribution: connecting it to the QUBO landscape via OGP + low-degree hardness + computational evidence (Table~\ref{tab:stability}).
\end{remark}

\begin{definition}[Coarse-grained dynamics]\label{def:gmacro}
Let $\{m_1, \ldots, m_k\}$ be the local minima of $H_\phi$. Define $g_{\text{macro}}(i) = \text{argmax}_j \Pr[\text{greedy-descent from } \tilde{m}_i \text{ reaches } m_j]$, where $\tilde{m}_i$ is obtained by flipping $\lceil\sqrt{n}\rceil$ random spins.
\end{definition}

\subsection{Falsification Criteria}

The proof would be \textbf{falsified} if any of the following are observed:

\begin{enumerate}
\item \textbf{Algorithm discovery}: A polynomial-time algorithm (of ANY type---stable, unstable, quantum, algebraic) that finds QUBO ground states with probability $> 1/\text{poly}(n)$ at $r_c$.

\item \textbf{Saturation failure}: Restart saturation breaks down at some $n^* > 50{,}000$---i.e., the landscape ``smooths out'' at large scale, with $\mathcal{N}(n) = \text{poly}(n)$ eventually.

\item \textbf{OGP collapse}: The forbidden overlap region gains non-negligible mass at large $n$---i.e., intermediate overlaps appear, providing algorithmic ``stepping stones''.

\item \textbf{RSB reversal}: $q_{\text{EA}} \to 0$ at large $n$, indicating the glass phase is a finite-size effect.

\item \textbf{Unstable advantage}: An input-unstable solver systematically outperforms stable solvers by more than $O(1/n)$ margin, suggesting instability helps.

\item \textbf{SOS violation}: A polynomial-time algorithm that cannot be captured by bounded-degree SOS relaxations, contradicting the low-degree hardness framework.
\end{enumerate}

None of these have been observed at any tested scale ($n \leq 50{,}000$, 15 solvers, 10 problem types).
